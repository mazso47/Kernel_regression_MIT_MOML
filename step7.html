def kernel3_relu_ntk(x, x_hat):
    xi = np.dot(x, x_hat).clip(min=-1.0, max= 1.0)      # By the definition of the dual
    ntk = (1/np.pi) * (xi * ((np.pi - np.arccos(xi)) + np.sqrt(1 - xi**2))) + xi * (1/np.pi) * (np.pi - np.arccos(xi))         # NTK of relu
    return ntk

def kernel_pred(x_train, alpha, kernel_func, x_new):
    kernel_vector = np.zeros((len(x_train)))

    for i in range(len(x_train)):
        kernel_vector[i] = kernel_func(x_train[i], x_new)
    return np.dot(alpha, kernel_vector)

class ScalingLayer(tf.keras.layers.Layer):                                                 # The scaling is added to the model as a custom scaling layer that has no trainable parameters
    def __init__(self, scale_factor, **kwargs):
        super(ScalingLayer, self).__init__(**kwargs)
        self.scale_factor = scale_factor

    def call(self, inputs):
        return inputs * self.scale_factor

def custom_mse(y_true, y_pred):                                                         # Custom loss function with the 0.5 scalar
    return 0.5 * tf.reduce_sum(tf.square(y_true - y_pred))


def train_and_test_model(hidden_layer_width):                                                       # Function to create and train the model
    scale_factor = tf.sqrt(2.0) / tf.sqrt(tf.cast(hidden_layer_width, tf.float32))                  # the scalar of sqrt(2)/sqrt(k) for the custom scaling layer

    model = tf.keras.Sequential([
        tf.keras.layers.Input(shape=(1,)),                                                          # Input layer
        tf.keras.layers.Dense(                                                                        # B with relu, so φ(B)
            units=hidden_layer_width,
            activation='relu',
            kernel_initializer=tf.keras.initializers.RandomNormal(mean=0.0, stddev=1.0)
        ),
        ScalingLayer(scale_factor),                                                                   # sqrt(2)/sqrt(k)
        tf.keras.layers.Dense(                                                                          # A
            activation = None,
            units=1,
            kernel_initializer=tf.keras.initializers.RandomNormal(mean=0.0, stddev=1.0)
        )
    ])

    optimizer = tf.keras.optimizers.SGD(learning_rate=0.01)
    model.compile(optimizer=optimizer, loss=custom_mse)

    #if hidden_layer_width >= 15:                                                                           #  I tried using the early stopping parameters recommended in the lecture but I could not get it to work well
     #  early_stopping = tf.keras.callbacks.EarlyStopping(monitor='mse', patience=1, restore_best_weights=True, min_delta=2e-3)
    #else:
     #  early_stopping = tf.keras.callbacks.EarlyStopping(monitor='mse', patience=50, restore_best_weights=True, min_delta=0)

    history = model.fit(x_train, y_train, epochs=30, batch_size=len(x_train), verbose=0)  # Training
    #callbacks=[early_stopping]

    X_test = np.array([0.1])                                                                                # Test the trained model on new datapoint
    y_pred = model.predict(X_test)

    return model, y_pred, history
